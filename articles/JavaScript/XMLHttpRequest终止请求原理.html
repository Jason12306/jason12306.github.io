XMLHttpRequest з»ҲжӯўиҜ·жұӮеҺҹзҗҶ вҖӢ

XMLHttpRequest.abort() ж–№жі•з”ЁдәҺз»ҲжӯўжӯЈеңЁиҝӣиЎҢзҡ„ HTTP иҜ·жұӮпјҢе…¶еә•еұӮеҺҹзҗҶж¶үеҸҠжөҸи§ҲеҷЁеј•ж“Һзҡ„еӨҡеұӮеҚҸдҪңгҖӮд»ҘдёӢжҳҜиҜҰз»ҶжҠҖжңҜи§Јжһҗпјҡ

ж ёеҝғеҺҹзҗҶ вҖӢ

  1. дёӯж–ӯзҪ‘з»ңеұӮиҝһжҺҘ

    • жөҸи§ҲеҷЁеҶ…ж ёзҡ„зҪ‘з»ңжЁЎеқ—ж”¶еҲ°з»ҲжӯўжҢҮд»ӨеҗҺпјҡ
      • иӢҘиҜ·жұӮд»ҚеңЁ TCP жҸЎжүӢйҳ¶ж®өпјҡз«ӢеҚіе…ій—ӯеҘ—жҺҘеӯ—пјҲsocketпјү
      • иӢҘе·Іе»әз«ӢиҝһжҺҘпјҡеҸ‘йҖҒ TCP RST еҢ…пјҲResetпјүејәеҲ¶ж–ӯејҖиҝһжҺҘ
      • иӢҘдҪҝз”Ё HTTP/2/3пјҡи§ҰеҸ‘ QUIC_RESET_STREAM жҲ– RST_STREAM её§

  2. жё…зҗҶиҜ·жұӮиө„жәҗ

    javascript
    // еҶ…йғЁдјӘд»Јз Ғе®һзҺ°
void XMLHttpRequest::abort() {
  if (m_request) {
    m_request->cancel();   // йҖҡзҹҘзҪ‘з»ңж Ҳз»Ҳжӯў
    m_request = nullptr;   // йҮҠж”ҫиҜ·жұӮеҜ№иұЎ
  }
  m_responseArrayBuffer = null; // жё…з©әжҺҘж”¶зј“еҶІеҢә
  transitionState(UNSENT);      // зҠ¶жҖҒеӣһйҖҖ
}

  3. зҠ¶жҖҒжңәйҮҚзҪ®

    • е°Ҷ XHR зҠ¶жҖҒејәеҲ¶еӣһйҖҖеҲ° UNSENT (0)
    • жё…з©әе·ІжҺҘж”¶зҡ„е“Қеә”еӨҙе’Ңе“Қеә”дҪ“ж•°жҚ®

дәӢд»¶и§ҰеҸ‘жөҒзЁӢ вҖӢ

е…ій”®зү№жҖ§ вҖӢ

  1. еҗҢжӯҘ/ејӮжӯҘе·®ејӮ

    • ејӮжӯҘиҜ·жұӮпјҡз«ӢеҚіз»ҲжӯўпјҢйҖҡиҝҮдәӢд»¶еӣһи°ғйҖҡзҹҘ
    • еҗҢжӯҘиҜ·жұӮпјҡйҳ»еЎһзҡ„ send() ж–№жі•дјҡжҠӣеҮә InvalidStateError ејӮеёё

  2. ж•°жҚ®е®Ңж•ҙжҖ§дҝқжҠӨ

    javascript
    xhr.onprogress = (e) => {
  if (e.lengthComputable) {
    console.log(`е·ІжҺҘж”¶: ${e.loaded}/${e.total} bytes`)
  }
}
xhr.abort() // з«ӢеҚіеҒңжӯўжҺҘж”¶ж•°жҚ®пјҢе·ІеҠ иҪҪйғЁеҲҶиў«дёўејғ

  3. еҶ…еӯҳз®ЎзҗҶ

    • з»ҲжӯўеҗҺе“Қеә”еҚ з”Ёзҡ„еҶ…еӯҳдјҡиў«ж Үи®°дёәеҸҜеӣһж”¶
    • еә•еұӮ ArrayBuffer еј•з”Ёиў«жё…йҷӨпјҢйҒҝе…ҚеҶ…еӯҳжі„жјҸ

дёҺ Fetch API AbortController зҡ„еҢәеҲ« вҖӢ

зү№жҖ§XMLHttpRequest.abort()Fetch + AbortController
дёӯж–ӯзІ’еәҰж•ҙдёӘ XHR еҜ№иұЎеҚ•дёӘ fetch иҜ·жұӮ
й”ҷиҜҜзұ»еһӢи§ҰеҸ‘ onerrorжҠӣеҮә AbortError ејӮеёё
е№¶еҸ‘жҺ§еҲ¶йңҖжүӢеҠЁз®ЎзҗҶе®һдҫӢе…ұдә« AbortSignal еҸҜжү№йҮҸз»Ҳжӯў
еҶ…еӯҳйҮҠж”ҫйңҖзҪ® null и§ЈйҷӨеј•з”ЁController ж–ӯејҖеҚійҮҠж”ҫ

е®һйҷ…еә”з”ЁеңәжҷҜ вҖӢ

javascript
// иҜ·жұӮи¶…ж—¶иҮӘеҠЁз»Ҳжӯў
const xhr = new XMLHttpRequest()
const timeoutId = setTimeout(() => {
  xhr.abort()
  console.log('Request timed out')
}, 5000)

xhr.open('GET', 'https://api.example.com/data')
xhr.onload = () => {
  clearTimeout(timeoutId)
  // еӨ„зҗҶж•°жҚ®
}
xhr.send()

  1. жёІжҹ“иҝӣзЁӢпјҡжү§иЎҢ JavaScript и°ғз”Ё abort()
  2. зҪ‘з»ңжңҚеҠЎ (Chrome)пјҡnetwork_service.mojom жҺҘеҸЈеӨ„зҗҶз»ҲжӯўжҢҮд»Ө
  3. еҘ—жҺҘеӯ—з®ЎзҗҶеҷЁпјҡж Үи®° socket дёә aborted зҠ¶жҖҒ
  4. зЈҒзӣҳзј“еӯҳпјҡдёӯж–ӯеҸҜиғҪеӯҳеңЁзҡ„зј“еӯҳеҶҷе…Ҙж“ҚдҪң
  5. еһғеңҫеӣһж”¶пјҡV8 еј•ж“ҺеңЁдёӢдёӘ GC е‘Ёжңҹеӣһж”¶зӣёе…іеҜ№иұЎ
